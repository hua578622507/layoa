CREATE DATABASE LAYOA;

#创建角色表
CREATE TABLE TB_ROLE(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT'主键',
	ROLE_CODE VARCHAR(30) COMMENT'角色编号',
	ROLE_NAME VARCHAR(30) COMMENT'角色名称',
	ROLE_KIND INT COMMENT'角色类型 #1:超级角色；0：普通角色',
	ROLE_INFO VARCHAR(30) COMMENT'角色简介',
	ACTIVE_FLAG INT DEFAULT '1' COMMENT'数据是否有效#1有效，0无效',
	CREATE_BY VARCHAR(30) COMMENT'创建人',
	CREATE_DATE DATETIME COMMENT'创建时间',
	UPDATE_BY VARCHAR(30) COMMENT'更新人',
	UPDATE_DATE DATETIME COMMENT'更新日期'
)
